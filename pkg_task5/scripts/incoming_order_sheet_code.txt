function doGet(e){
  
  var ss = SpreadsheetApp.getActive();

  var sheet = ss.getSheetByName(e.parameter["Id"]);

  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];

  var lastRow = sheet.getLastRow();

  var cell = sheet.getRange('a1');
  var col = 0;
  var d = new Date();

  for (i in headers){

    // loop through the headers and if a parameter name matches the header name insert the value

    if (headers[i] == "Timestamp")
    {
      val = d.toDateString() + ", " + d.toLocaleTimeString();
    }
    else if(headers[i]=="Team Id")
    {
      val = e.parameter["Team_Id"]; 
    }
    else if(headers[i]=="Unique Id")
    {
      val = e.parameter["Unique_Id"]; 

    }
    else if(headers[i]=="Order Date and Time")
    {
      val = e.parameter["Order_Date_and_Time"]; 
    }
    else if(headers[i]=="Order Quantity")
    {
      val = e.parameter["Order_Quantity"]; 
    }
    else if(headers[i]=="Order Id")
    {
      val = e.parameter["Order_Id"]; 
    }

    else
    {
      val = e.parameter[headers[i]];
    }
    
    // append data to the last row
    cell.offset(lastRow, col).setValue(val);
    col++;
  }

  return ContentService.createTextOutput('success');
}
